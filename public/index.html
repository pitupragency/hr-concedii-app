function toLocalISODate(date){ const year = date.getFullYear() const month = String(date.getMonth() + 1).padStart(2, '0') const day = String(date.getDate()).padStart(2, '0') return `${year}-${month}-${day}` } function calcDays(a, b){ if(!a || !b) return '' const start = new Date(`${a}T12:00:00`) const end = new Date(`${b}T12:00:00`) if(Number.isNaN(start.getTime()) || Number.isNaN(end.getTime())) return '' if(end < start) return '' let count = 0 const current = new Date(start) while(current <= end){ const dayOfWeek = current.getDay() const isoDate = toLocalISODate(current) const isWeekend = dayOfWeek === 0 || dayOfWeek === 6 const isPublicHoliday = ROMANIA_PUBLIC_HOLIDAYS_2026.has(isoDate) if(!isWeekend && !isPublicHoliday){ count++ } current.setDate(current.getDate() + 1) } return count } function updateTotalDaysField(){ const startValue = document.getElementById('startDate').value const endValue = document.getElementById('endDate').value const result = calcDays(startValue, endValue) document.getElementById('totalDays').value = result === '' ? '' : result } document.getElementById('startDate').addEventListener('change', updateTotalDaysField) document.getElementById('endDate').addEventListener('change', updateTotalDaysField) document.getElementById('requestForm').addEventListener('submit', async e => { e.preventDefault() hide('appError'); hide('appSuccess') try { updateTotalDaysField() const payload = { employee_id: profile.id, leave_type_id: document.getElementById('leaveTypeSelect').value || null, start_date: document.getElementById('startDate').value, end_date: document.getElementById('endDate').value, total_days: Number(document.getElementById('totalDays').value || 0), note: document.getElementById('requestNote').value.trim() || null } const { error } = await supabase.from('leave_requests').insert(payload) if(error) throw error document.getElementById('requestForm').reset() document.getElementById('totalDays').value = '' show('appSuccess', 'The request was submitted successfully.') await refreshAll() } catch(err) { console.error(err) show('appError', 'Could not save the request: ' + (err.message || JSON.stringify(err)) + ' — check the INSERT policy on leave_requests and leave_types.') } })
